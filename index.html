function validateForm(input) {
   const errors = [];
   if (!input.email.includes('@')) {
     errors.push('Invalid email');
   }
   if (input.password.length < 8) {
     errors.push('Password too short');
   }
   return errors.length === 0
     ? 'All good!' : errors.join(', ');
}
